	.ol-zoom {
		top : .5em;
	}
	
	SECTION.mainView .mapBtnArea .mapBtnGroup {
	    top: 5.5em;
	}
</style>
<script type="text/javascript">
	var map = null;
	window.onload = function() {
		map = new JBMap({
			isRotate : true,
			baseLayers : {
				"BASE" : {
					layer : [
					    new ol.layer.Tile({
							source : new ol.source.OSM()
						})
					]
				}
			}
		});

		document.getElementById("measureLine").onclick = function() {
			var bool = this.className.indexOf("on") == -1;
			
			var areaBtn = document.getElementById("measurePolygon");
			if(areaBtn.className.indexOf("on") != -1) {
				areaBtn.click();
			}

			map.setMeasureControl("NONE");
			if(bool) {
				this.className = "mapEditor on";
				map.setMeasureControl("LENGTH");
			}else{
				this.className = "mapEditor";
			}
		};

		document.getElementById("measurePolygon").onclick = function() {
			var bool = this.className.indexOf("on") == -1;

			var lengthBtn = document.getElementById("measureLine");
			if(lengthBtn.className.indexOf("on") != -1) {
				lengthBtn.click();
			}

			map.setMeasureControl("NONE");
			if(bool) {
				this.className = "mapEditor on";
				map.setMeasureControl("AREA");
			}else{
				this.className = "mapEditor";
			}
		};
	}
</script>
